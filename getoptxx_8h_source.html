297 inline auto getoptxx::v1::arguments::parse(int argc, char* const argv[],
298  std::initializer_list<option> options) -> arguments {
299  arguments args;
300  auto const ae = std::find_if(argv+1, argv+argc,
301  [] (key_type const& s) { return s == "--"; });
302 
303  for (auto&& a = argv+1; a != ae && !args.help(); ++a) {
304  if (!*a) continue; // ignore empty arguments
305 
306  if ((*a)[0] != '-') { // non-option arguments
307  args.m_unparsed.push_back(*a);
308  continue;
309  } else if (!(*a)[1]) { // ignore a single '-'
310  continue;
311  }
312 
313  auto const s = key_type{((*a)[1] == '-') ? &(*a)[2] : &(*a)[1]};
314  args.m_help = (s == "h" || s == "help");
315  if (args.m_help) return args;
316 
317  auto const o = std::find_if(std::begin(options), std::end(options),
318  [&s] (auto o) { return s == o.m_short || s == o.m_long; });
319  if (o == std::end(options)) {
320  auto const what = "unknown option '"+s.to_string()+"'";
321  throw std::runtime_error{std::move(what)};
322  }
323 
324  auto const v = [&s,&o,&a,&ae] () -> value_type {
325  if (o->m_aflags == option::argument_flags::required ||
326  o->m_aflags == option::argument_flags::optional) {
327  if (a+1 < ae && (a+1 && (*(a+1))[0] != '-')) {
328  return *(++a);
329  } else if (o->m_aflags == option::argument_flags::required) {
330  auto const what =
331  "option '"+s.to_string()+"' requires a value";
332  throw std::runtime_error{std::move(what)};
333  }
334  }
335  return {};
336  }();
337 
338  if (!o->m_short.empty()) {
339  args.m_parsed.insert(std::make_pair(o->m_short, v));
340  }
341  if (!o->m_long.empty()) {
342  args.m_parsed.insert(std::make_pair(o->m_long, v));
343  }
344  }
345 
346  std::for_each(std::begin(options), std::end(options), [&args] (auto&& o) {
347  if (o.m_oflags != option::option_flags::required) return;
348 
349  if ((!o.m_short.empty() && args.m_parsed.count(o.m_short) == 0)||
350  (!o.m_long.empty() && args.m_parsed.count(o.m_long) == 0)) {
351  auto const what = "option '"+
352  (!o.m_long.empty()?o.m_long.to_string():o.m_short.to_string())+
353  "' required";
354  throw std::runtime_error{std::move(what)};
355  }
356  });
357 
358  args.m_unparsed.insert(std::end(args.m_unparsed), ae+1, argv+argc);
359  return args;
360 }
